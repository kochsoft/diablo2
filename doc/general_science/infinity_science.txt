def bin2bts(bin, pfname="/home/kochmn/ks.cube"):

    while len(bin) % 8 != 0:
        bin = bin + '0'
    n = len(bin)//8
    val = int(bin[::-1],2)
    bts = val.to_bytes(n,'little')
